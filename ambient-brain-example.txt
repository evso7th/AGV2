// ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ –¢–ò–ü–´ ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
interface AmbientBrainConfig {
  durationSec?: number; // –æ–±—â–∞—è –¥–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Ç—Ä–µ–∫–∞ (–¥–ª—è –∞–≤—Ç–æ–ø–∏–ª–æ—Ç–∞)
  moodTrajectory?: 'static' | 'wave' | 'ascent' | 'descent'; // —Ç—Ä–∞–µ–∫—Ç–æ—Ä–∏—è –Ω–∞—Å—Ç—Ä–æ–µ–Ω–∏—è
}

interface AmbientBrainState {
  timeSec: number;
  fog: number;    // 0..1 ‚Äî —Å–ø–µ–∫—Ç—Ä–∞–ª—å–Ω–∞—è –ø–ª–æ—Ç–Ω–æ—Å—Ç—å (—Ç—É–º–∞–Ω)
  pulse: number;  // 0..1 ‚Äî —Ä–∏—Ç–º–∏—á–µ—Å–∫–∞—è –∂–∏–∑–Ω—å (–ø—É–ª—å—Å–∞—Ü–∏—è)
  depth: number;  // 0..1 ‚Äî –≥–∞—Ä–º–æ–Ω–∏—á–µ—Å–∫–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å
  harmonicState: {
    centerMidi: number;   // —Ç–µ–∫—É—â–∏–π –≥–∞—Ä–º–æ–Ω–∏—á–µ—Å–∫–∏–π —Ü–µ–Ω—Ç—Ä (–Ω–∞–ø—Ä–∏–º–µ—Ä, 55 = –ª—è)
    mode: 'dorian' | 'lydian' | 'phrygian' | 'mixolydian' | 'aeolian';
    density: number;      // 1..4 ‚Äî –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–ª–æ—ë–≤ –≤ –∞–∫–∫–æ—Ä–¥–µ
    lastTransitionSec: number;
  };
  tensionWaves: number[]; // [T0, T1, T2, T3] ‚Äî 4 –º–∞—Å—à—Ç–∞–±–∞ —Ñ—Ä–∞–∫—Ç–∞–ª—å–Ω–æ–π –∫—Ä–∏–≤–æ–π
}

// ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ –§–†–ê–ö–¢–ê–õ–¨–ù–ê–Ø –ö–†–ò–í–ê–Ø –ù–ê–ü–†–Ø–ñ–ï–ù–ò–Ø (–≥–æ—Ç–æ–≤—ã–π —Ä–µ—Ü–µ–ø—Ç) ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
const PHI = [0.0, 1.309, 2.618, 4.188];   // œÜ‚ÇÄ..œÜ‚ÇÉ ‚Äî –∑–æ–ª–æ—Ç–æ–µ —Å–µ—á–µ–Ω–∏–µ √ó ‚àö2
const PSI = [0.785, 2.094, 3.927, 5.498]; // œà‚ÇÄ..œà‚ÇÉ ‚Äî œÄ/4, 2œÄ/3, 5œÄ/4, 7œÄ/4 + –¥–∂–∏—Ç—Ç–µ—Ä
const BETA = 7.2360679775;                // –∏—Ä—Ä–∞—Ü–∏–æ–Ω–∞–ª—å–Ω—ã–π –º–Ω–æ–∂–∏—Ç–µ–ª—å
const PERIODS = [8, 45, 180, 600];        // –º–∞—Å—à—Ç–∞–±—ã –≤—Ä–µ–º–µ–Ω–∏ –≤ —Å–µ–∫—É–Ω–¥–∞—Ö
const WEIGHTS = [0.15, 0.35, 0.30, 0.20]; // –≤–µ—Å–∞ –º–∞—Å—à—Ç–∞–±–æ–≤
const MOD_DEPTH = [0.05, 0.12, 0.25, 0.40]; // –≥–ª—É–±–∏–Ω–∞ –º–æ–¥—É–ª—è—Ü–∏–∏

function computeTensionWaves(timeSec: number): number[] {
  const waves = [];
  for (let n = 0; n < 4; n++) {
    const base = Math.sin(2 * Math.PI * timeSec / PERIODS[n] + PHI[n]);
    const mod = Math.sin(2 * Math.PI * timeSec / (PERIODS[n] / BETA) + PSI[n]);
    const wave = base * (1 + MOD_DEPTH[n] * mod);
    waves.push(wave);
  }
  return waves;
}

function computeGlobalTension(waves: number[]): number {
  let tension = 0;
  for (let n = 0; n < 4; n++) tension += WEIGHTS[n] * waves[n];
  return (tension + 1) / 2; // –Ω–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è –≤ [0..1]
}

// ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ –ú–û–î–ê–õ–¨–ù–´–ï –ü–†–ê–í–ò–õ–ê –ì–ê–†–ú–û–ù–ò–ò ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
const MODE_INTERVALS = {
  dorian: [0, 2, 3, 5, 7, 9, 10],
  lydian: [0, 2, 4, 6, 7, 9, 11],
  phrygian: [0, 1, 3, 5, 7, 8, 10],
  mixolydian: [0, 2, 4, 5, 7, 9, 10],
  aeolian: [0, 2, 3, 5, 7, 8, 10]
} as const;

function generateAmbientChord(centerMidi: number, mode: keyof typeof MODE_INTERVALS, density: number): number[] {
  const intervals = MODE_INTERVALS[mode];
  const notes = [centerMidi - 12]; // —Å—É–±-–±–∞—Å (–æ–∫—Ç–∞–≤–∞ -1)
  
  // –í–µ—Ä—Ö–Ω–∏–µ –≥–æ–ª–æ—Å–∞: —Ä–∞—Å–ø—Ä–µ–¥–µ–ª—è–µ–º –ø–æ —Ä–µ–≥–∏—Å—Ç—Ä–∞–º
  for (let i = 1; i <= Math.min(density, intervals.length - 1); i++) {
    const interval = intervals[i];
    if (i <= 2) {
      notes.push(centerMidi + interval); // –ø–∞–¥ (–æ–∫—Ç–∞–≤–∞ 0)
    } else {
      notes.push(centerMidi + interval + 12); // —Å–∏—è–Ω–∏–µ (–æ–∫—Ç–∞–≤–∞ +1)
    }
  }
  
  return notes.filter(n => n >= 36 && n <= 96);
}

// ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ –ú–ê–†–ö–û–í–°–ö–ê–Ø –ú–ê–®–ò–ù–ê –ü–ï–†–ï–•–û–î–û–í ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
const MODE_TRANSITIONS = {
  dorian: ['dorian', 'lydian', 'aeolian'],
  lydian: ['lydian', 'dorian', 'mixolydian'],
  phrygian: ['phrygian', 'aeolian', 'dorian'],
  mixolydian: ['mixolydian', 'lydian', 'dorian'],
  aeolian: ['aeolian', 'phrygian', 'dorian']
} as const;

function getNextMode(currentMode: keyof typeof MODE_TRANSITIONS, depth: number): keyof typeof MODE_TRANSITIONS {
  const candidates = MODE_TRANSITIONS[currentMode];
  // –ü—Ä–∏ –≤—ã—Å–æ–∫–æ–º depth ‚Äî —á–∞—â–µ –º–µ–Ω—è–µ–º –º–æ–¥–∞–ª—å–Ω–æ—Å—Ç—å
  if (Math.random() < depth * 0.6) {
    return candidates[Math.floor(Math.random() * candidates.length)] as keyof typeof MODE_TRANSITIONS;
  }
  return currentMode;
}

// ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ –û–°–ù–û–í–ù–û–ô –ö–õ–ê–°–° –ú–û–ó–ì–ê ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
export class AmbientBrain {
  private state: AmbientBrainState;
  private config: AmbientBrainConfig;
  private instruments: Record<string, any> = {};
  private lastChord: number[] = [];
  private goosebumpCooldown = 0; // –∑–∞—â–∏—Ç–∞ –æ—Ç —á–∞—Å—Ç—ã—Ö "–º—É—Ä–∞—à–µ–∫"

  constructor(config: AmbientBrainConfig = {}) {
    this.config = { durationSec: 420, moodTrajectory: 'wave', ...config };
    this.state = {
      timeSec: 0,
      fog: 0.3,
      pulse: 0.15,
      depth: 0.4,
      harmonicState: {
        centerMidi: 55, // –ª—è
        mode: 'dorian',
        density: 2,
        lastTransitionSec: 0
      },
      tensionWaves: [0, 0, 0, 0]
    };
  }

  // –†–µ–≥–∏—Å—Ç—Ä–∞—Ü–∏—è –∏–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç–æ–≤ –¥–ª—è —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è
  registerInstrument(name: string, instrument: any) {
    this.instruments[name] = instrument;
    console.log(`[AmbientBrain] Registered ${name}`);
  }

  // –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ —Å–æ—Å—Ç–æ—è–Ω–∏—è (–≤—ã–∑—ã–≤–∞—Ç—å –∫–∞–∂–¥—É—é —Å–µ–∫—É–Ω–¥—É –≤ –∫–æ–º–ø–æ–∑–µ—Ä–µ)
  update(ctx: AudioContext, timeSec: number) {
    this.state.timeSec = timeSec;
    this.state.tensionWaves = computeTensionWaves(timeSec);
    const globalTension = computeGlobalTension(this.state.tensionWaves);

    // ‚îÄ‚îÄ‚îÄ 1. –ê–í–¢–û–ü–ò–õ–û–¢ –ù–ê–°–¢–†–û–ï–ù–ò–Ø ‚îÄ‚îÄ‚îÄ
    this.updateMoodAxes(timeSec);

    // ‚îÄ‚îÄ‚îÄ 2. –ì–ê–†–ú–û–ù–ò–ß–ï–°–ö–ò–ï –ü–ï–†–ï–•–û–î–´ ‚îÄ‚îÄ‚îÄ
    this.updateHarmony(ctx, timeSec, globalTension);

    // ‚îÄ‚îÄ‚îÄ 3. –ü–†–ò–ú–ï–ù–ï–ù–ò–ï –ö –ò–ù–°–¢–†–£–ú–ï–ù–¢–ê–ú ‚îÄ‚îÄ‚îÄ
    this.applyToInstruments(ctx, timeSec);

    // ‚îÄ‚îÄ‚îÄ 4. –≠–§–§–ï–ö–¢ "–ú–£–†–ê–®–ï–ö" (–∫–æ–Ω—Ç—Ä–∞—Å—Ç —Ç—É–º–∞–Ω/—á–∏—Å—Ç–æ—Ç–∞ + –º–∏–∫—Ä–æ-—Å–¥–≤–∏–≥) ‚îÄ‚îÄ‚îÄ
    this.triggerGoosebumps(ctx, timeSec);
  }

  private updateMoodAxes(timeSec: number) {
    const progress = Math.min(timeSec / (this.config.durationSec || 420), 1.0);
    
    switch (this.config.moodTrajectory) {
      case 'wave':
        this.state.fog = 0.2 + 0.6 * Math.sin(progress * Math.PI);
        this.state.pulse = 0.1 + 0.4 * (1 - Math.cos(progress * 2 * Math.PI));
        this.state.depth = 0.3 + 0.5 * progress;
        break;
      case 'ascent':
        this.state.fog = 0.1 + 0.3 * progress;
        this.state.pulse = 0.05 + 0.25 * progress;
        this.state.depth = 0.2 + 0.7 * progress;
        break;
      case 'descent':
        this.state.fog = 0.8 - 0.6 * progress;
        this.state.pulse = 0.6 - 0.45 * progress;
        this.state.depth = 0.9 - 0.6 * progress;
        break;
      default: // static
        this.state.fog = 0.4;
        this.state.pulse = 0.2;
        this.state.depth = 0.5;
    }
  }

  private updateHarmony(ctx: AudioContext, timeSec: number, tension: number) {
    const { harmonicState } = this.state;
    const timeSinceLast = timeSec - harmonicState.lastTransitionSec;
    
    // –¢—Ä–∏–≥–≥–µ—Ä –ø–µ—Ä–µ—Ö–æ–¥–∞: —Ä–µ–∑–∫–∏–π —Å–∫–∞—á–æ–∫ –Ω–∞–ø—Ä—è–∂–µ–Ω–∏—è + –º–∏–Ω–∏–º–∞–ª—å–Ω–∞—è –ø–∞—É–∑–∞
    const tensionDelta = Math.abs(tension - computeGlobalTension(this.state.tensionWaves.map(w => w * 0.9)));
    const shouldTransition = tensionDelta > 0.22 && timeSinceLast > 45 && Math.random() < this.state.depth * 0.7;
    
    if (shouldTransition) {
      // –°–º–µ–Ω–∞ —Ü–µ–Ω—Ç—Ä–∞ —á–µ—Ä–µ–∑ –∑–æ–ª–æ—Ç–æ–µ —Å–µ—á–µ–Ω–∏–µ (–µ—Å—Ç–µ—Å—Ç–≤–µ–Ω–Ω–∞—è –Ω–µ–ø—Ä–µ–¥—Å–∫–∞–∑—É–µ–º–æ—Å—Ç—å)
      const goldenShift = Math.round(7.8 * (Math.random() > 0.5 ? 1 : -1)); // ‚âà –ø–æ–ª—É—Ç–æ–Ω–æ–≤
      harmonicState.centerMidi = clamp(harmonicState.centerMidi + goldenShift, 48, 72);
      
      // –°–º–µ–Ω–∞ –º–æ–¥–∞–ª—å–Ω–æ—Å—Ç–∏
      harmonicState.mode = getNextMode(harmonicState.mode, this.state.depth);
      
      // –ü–ª–æ—Ç–Ω–æ—Å—Ç—å = —Ñ—É–Ω–∫—Ü–∏—è –æ—Ç –≥–ª—É–±–∏–Ω—ã
      harmonicState.density = Math.floor(1 + this.state.depth * 3);
      
      harmonicState.lastTransitionSec = timeSec;
      
      // –í–æ—Å–ø—Ä–æ–∏–∑–≤–µ–¥–µ–Ω–∏–µ –Ω–æ–≤–æ–≥–æ –∞–∫–∫–æ—Ä–¥–∞
      const newChord = generateAmbientChord(
        harmonicState.centerMidi,
        harmonicState.mode,
        harmonicState.density
      );
      
      // –ü–ª–∞–≤–Ω–æ–µ –∑–∞—Ç—É—Ö–∞–Ω–∏–µ —Å—Ç–∞—Ä–æ–≥–æ –∞–∫–∫–æ—Ä–¥–∞ + –≤—Å—Ç—É–ø–ª–µ–Ω–∏–µ –Ω–æ–≤–æ–≥–æ
      if (this.instruments.pad) {
        this.lastChord.forEach(midi => this.instruments.pad.noteOff(midi, ctx.currentTime));
        setTimeout(() => {
          newChord.forEach(midi => this.instruments.pad.noteOn(midi, ctx.currentTime, 0.8));
        }, 300);
      }
      
      this.lastChord = newChord;
      console.log(`[AmbientBrain] Harmony shift ‚Üí center=${harmonicState.centerMidi} (${harmonicState.mode}), density=${harmonicState.density}`);
    }
  }

  private applyToInstruments(ctx: AudioContext, timeSec: number) {
    const { fog, pulse, depth } = this.state;
    const [T0, T1, T2, T3] = this.state.tensionWaves;
    
    // ‚îÄ‚îÄ‚îÄ PAD: —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ —Ñ–∏–ª—å—Ç—Ä–æ–º, —Ä–µ–≤–µ—Ä–±–µ—Ä–∞—Ü–∏–µ–π, LFO ‚îÄ‚îÄ‚îÄ
    if (this.instruments.pad?.setParam) {
      // LPF cutoff: —Ç—É–º–∞–Ω = –∑–∞–∫—Ä—ã—Ç–∏–µ —Ñ–∏–ª—å—Ç—Ä–∞ (–Ω–µ–ª–∏–Ω–µ–π–Ω–æ)
      const cutoff = 1400 * (1 - Math.pow(fog, 1.5));
      this.instruments.pad.setParam('lpfCutoff', cutoff);
      
      // –†–µ–≤–µ—Ä–±–µ—Ä–∞—Ü–∏—è: —Ç—É–º–∞–Ω = –±–æ–ª—å—à–µ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞
      // (–≤ —Ç–µ–∫—É—â–µ–π —Ñ–∞–±—Ä–∏–∫–µ –Ω–µ—Ç –ø—Ä—è–º–æ–≥–æ —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è ‚Äî –∏–º–∏—Ç–∏—Ä—É–µ–º —á–µ—Ä–µ–∑ –≥—Ä–æ–º–∫–æ—Å—Ç—å —Ä–µ–≤–µ—Ä–±–∞)
      // –í –ø—Ä–æ–¥–∞–∫—à–µ–Ω–µ: —Ä–∞—Å—à–∏—Ä–∏—Ç—å —Å–∏–Ω—Ç –¥–ª—è —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è `reverbMix` —á–µ—Ä–µ–∑ –ø–∞—Ä–∞–º–µ—Ç—Ä
    }
    
    // ‚îÄ‚îÄ‚îÄ –ë–ê–°: –ø–ª–∞–≤–Ω—ã–π –¥—Ä–µ–π—Ñ —á–∞—Å—Ç–æ—Ç—ã –ø—Ä–∏ –≤—ã—Å–æ–∫–æ–º —Ç—É–º–∞–Ω–µ ‚îÄ‚îÄ‚îÄ
    if (this.instruments.bass && T3 > 0.7 && timeSec - this.state.harmonicState.lastTransitionSec > 20) {
      // –ú–∏–∫—Ä–æ-–¥—Ä–µ–π—Ñ –±–∞—Å–∞: ¬±0.3 –ø–æ–ª—É—Ç–æ–Ω–∞ –∑–∞ 30 —Å–µ–∫
      const drift = Math.sin(timeSec * 0.03) * 0.3;
      const driftedFreq = midiToHz(this.state.harmonicState.centerMidi + drift);
      // –í —Ç–µ–∫—É—â–µ–π –∞—Ä—Ö–∏—Ç–µ–∫—Ç—É—Ä–µ ‚Äî –ø–µ—Ä–µ–∑–∞–ø—É—Å–∫ –Ω–æ—Ç—ã (–ø–∞—Ç—á –≤—ã—à–µ –¥–∞—ë—Ç –±–∞–∑–æ–≤—É—é —Ñ—É–Ω–∫—Ü–∏–æ–Ω–∞–ª—å–Ω–æ—Å—Ç—å)
    }
    
    // ‚îÄ‚îÄ‚îÄ –û–†–ì–ê–ù: —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ —Å–ª–æ–∂–Ω–æ—Å—Ç—å—é drawbar'–æ–≤ —á–µ—Ä–µ–∑ –≥–ª—É–±–∏–Ω—É ‚îÄ‚îÄ‚îÄ
    if (this.instruments.organ && depth > 0.6) {
      // –ü—Ä–∏ –≤—ã—Å–æ–∫–æ–π –≥–ª—É–±–∏–Ω–µ ‚Äî –∞–∫—Ç–∏–≤–∏—Ä—É–µ–º –±–æ–ª—å—à–µ drawbar'–æ–≤
      // (–≤ –ø—Ä–æ–¥–∞–∫—à–µ–Ω–µ: —Ä–∞—Å—à–∏—Ä–∏—Ç—å organ engine –¥–ª—è –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–π –Ω–∞—Å—Ç—Ä–æ–π–∫–∏ drawbars)
    }
  }

  private triggerGoosebumps(ctx: AudioContext, timeSec: number) {
    const { fog, pulse, depth } = this.state;
    const [T0, T1, T2] = this.state.tensionWaves;
    
    // –£—Å–ª–æ–≤–∏–µ "–º—É—Ä–∞—à–µ–∫": –ø–∏–∫ —Ç—É–º–∞–Ω–∞ + —Å—Ç–∞—Ç–∏–∫–∞ + –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–∞—è –≥–ª—É–±–∏–Ω–∞
    const isGoosebumpMoment = 
      fog > 0.85 && 
      pulse < 0.15 && 
      depth > 0.6 && 
      T2 > 0.8 && 
      timeSec - this.goosebumpCooldown > 120; // –∫—É–ª–¥–∞—É–Ω 2 –º–∏–Ω—É—Ç—ã
    
    if (isGoosebumpMoment) {
      this.goosebumpCooldown = timeSec;
      
      console.log(`[AmbientBrain] üåä GOOSEBUMPS TRIGGERED at ${timeSec.toFixed(1)}s`);
      
      // 1. –†–µ–∑–∫–æ–µ —Å–Ω–∏–∂–µ–Ω–∏–µ —Ç—É–º–∞–Ω–∞ (–ø—Ä–æ—Ä—ã–≤ –≤ —á–∏—Å—Ç–æ—Ç—É)
      this.state.fog = 0.2;
      
      // 2. –ú–∏–∫—Ä–æ-—Å–¥–≤–∏–≥ –±–∞—Å–∞ –Ω–∞ +1 –ø–æ–ª—É—Ç–æ–Ω –∑–∞ 8 —Å–µ–∫
      if (this.instruments.bass?.setBaseFrequency) {
        this.instruments.bass.setBaseFrequency(this.state.harmonicState.centerMidi + 1, 8);
      }
      
      // 3. –í—Å–ø–ª–µ—Å–∫ —Ä–µ–≤–µ—Ä–±–µ—Ä–∞—Ü–∏–∏ –¥–ª—è "–æ—Ç–∫—Ä—ã—Ç–∏—è –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞"
      if (this.instruments.pad) {
        // –í –ø—Ä–æ–¥–∞–∫—à–µ–Ω–µ: —É–ø—Ä–∞–≤–ª—è—Ç—å reverbMix —á–µ—Ä–µ–∑ –ø–∞—Ä–∞–º–µ—Ç—Ä
        console.log('[AmbientBrain] Simulating reverb swell (requires engine patch)');
      }
    }
  }

  // –ü–æ–ª—É—á–µ–Ω–∏–µ —Ç–µ–∫—É—â–µ–≥–æ —Å–æ—Å—Ç–æ—è–Ω–∏—è –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏/–≤–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏–∏
  getState() {
    return { ...this.state };
  }
}
