  // –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Å–ª–µ–¥—É—é—â–∏–π –∞–∫–∫–æ—Ä–¥ –¥–ª—è –≤–µ–¥—É—â–µ–π –Ω–æ—Ç—ã

private getNextChordRoot(barIn12: number, currentChord: GhostChord): number {
  // 12-bar blues progression mapping
  const progression = [0, 0, 0, 0, 5, 5, 0, 0, 7, 5, 0, 7]; // offsets from root in semitones
  const nextBar = (barIn12 + 1) % 12;
  return currentChord.rootNote + progression[nextBar];
}
